## Install

```bash
npm install --save nextjs-jsoneditor
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'nextjs-jsoneditor'
import 'nextjs-jsoneditor/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
